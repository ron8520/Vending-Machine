package W14B_G4_Assignment2.model;
import java.util.*;

public class Transaction{

    private Date date;
    private List<String> items;
    private double money_paid;
    private double money_return;
    private Method method;

    public Transaction(Date date, List<String> items, double money_paid, double money_return, Method method){
        this.date = date;
        this.items = items;
        this.money_paid = money_paid;
        this.money_return = money_return;
        this.method = method;
    }

    public Date getDate(){
        return this.date;
    }

    public List<String> getItems(){
        return this.items;
    }

    public double getPaid(){
        return this.money_paid;
    }

    public double getChange(){
        return this.money_return;
    }

    public Method getMethod(){
        return this.method;
    }
}
